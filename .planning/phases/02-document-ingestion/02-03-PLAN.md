```csharp
using DocumentFormat.OpenXml.Packaging;
using DocumentFormat.OpenXml.Presentation;
using A = DocumentFormat.OpenXml.Drawing;

public record SlideContent(int SlideNumber, string FileName, string BodyText, string NotesText);

// Key: NotesSlidePart may be null — guard with null check
var slidePart = (SlidePart)presentationPart.GetPartById(relId);
var bodyText = string.Concat(slidePart.Slide.Descendants<A.Text>().Select(t => t.Text));
if (slidePart.NotesSlidePart is { } notesPart)
    notesText = string.Concat(notesPart.NotesSlide.Descendants<A.Text>().Select(t => t.Text));
```

<!-- PDF extraction pattern from RESEARCH.md — use GetWords() not raw Letters -->
```csharp
using UglyToad.PdfPig;

public record PageContent(int PageNumber, string FileName, string Text, bool NeedsVision);

// Use GetWords() for readable text (not raw Letters which can be garbled)
var words = page.GetWords();
var text = string.Join(" ", words.Select(w => w.Text));
bool hasText = words.Any();
// NeedsVision = !hasText (blank page detection: no words = image-only page)
```
