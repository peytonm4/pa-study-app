```tsx
import mermaid from 'mermaid';
import { useEffect, useId, useState } from 'react';

mermaid.initialize({ startOnLoad: false });

export function MermaidDiagram({ syntax }: { syntax: string }) {
  const id = useId().replace(/:/g, '');
  const [svg, setSvg] = useState('');
  useEffect(() => {
    let cancelled = false;
    mermaid.render(`m${id}`, syntax).then(({ svg }) => {
      if (!cancelled) setSvg(svg);
    });
    return () => { cancelled = true; };
  }, [syntax, id]);
  return <div dangerouslySetInnerHTML={{ __html: svg }} />;
}
```

```typescript
import client from './client';

export interface SectionSummary {
  id: string;
  headingText: string;
  sortOrder: number;
  hasConceptMap: boolean;
  cardsReviewed: number;
  quizCompleted: boolean;
}

export interface StudyGuide {
  directAnswer: string;
  highYieldDetails: string[];
  keyTables: string[];
  mustKnowNumbers: string[];
  sources: string[];
}

export interface Flashcard {
  id: string;
  front: string;
  back: string;
  cardType: 'qa' | 'cloze';
  sourceRefs: string[];
}

export interface QuizQuestion {
  id: string;
  questionText: string;
  choices: string[];
  correctAnswer: string;
  sourceRef: string;
}

export interface ConceptMap {
  mermaidSyntax: string;
}

export interface ProgressUpdate {
  cardsReviewed?: number;
  quizCompleted?: boolean;
}

export const study = {
  listSections: (moduleId: string) =>
    client.get<SectionSummary[]>(`/modules/${moduleId}/sections`).then(r => r.data),
  getStudyGuide: (sectionId: string) =>
    client.get<StudyGuide>(`/sections/${sectionId}/study-guide`).then(r => r.data),
  getFlashcards: (sectionId: string) =>
    client.get<Flashcard[]>(`/sections/${sectionId}/flashcards`).then(r => r.data),
  getQuiz: (sectionId: string) =>
    client.get<QuizQuestion[]>(`/sections/${sectionId}/quiz`).then(r => r.data),
  getConceptMap: (sectionId: string) =>
    client.get<ConceptMap>(`/sections/${sectionId}/concept-map`).then(r => r.data),
  upsertProgress: (sectionId: string, body: ProgressUpdate) =>
    client.post(`/sections/${sectionId}/progress`, body).then(r => r.data),
};
```

```tsx
import mermaid from 'mermaid';
import { useEffect, useId, useState } from 'react';

mermaid.initialize({ startOnLoad: false });

export function MermaidDiagram({ syntax }: { syntax: string }) {
  const rawId = useId().replace(/:/g, '');
  const [svg, setSvg] = useState('');
  useEffect(() => {
    let cancelled = false;
    mermaid.render(`m${rawId}`, syntax).then(({ svg }) => {
      if (!cancelled) setSvg(svg);
    });
    return () => { cancelled = true; };
  }, [syntax, rawId]);
  if (!svg) return <div className="text-muted-foreground text-sm">Rendering diagram...</div>;
  return <div dangerouslySetInnerHTML={{ __html: svg }} />;
}
```
